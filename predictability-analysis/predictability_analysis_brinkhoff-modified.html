def get_sequences_missing(users_traj,sequence_length = sequence_length):
    users = unique(users_traj['id'])
    print("Number of users: " + str(len(users)))
    sequences = dict()
    missing = dict()
    user_ind = users_traj.set_index(['id'])
    #print(user_trajs_ind.head)
    for user in users:
        sequences[user] = [-1] * sequence_length    
        sequence_locs = user_ind.loc[user][['t','r']]
        missing[user] = 1 - (float(len(sequence_locs))/sequence_length)
        #print ("Location count: " + str(len(sequence_locs)))
        #print ("Missing: " + str(missing[user]))
        times = np.array(sequence_locs['t']).astype(int)
        locs = np.array(sequence_locs['r']).astype(int)
        for i in range(len(times)):
            sequences[user][times[i]] = locs[i]
        #print(sequences[user])
    return [users,sequences,missing]

def calc_entropy_q(users,sequences,missing,threshold):
    entropy_q = pd.DataFrame(columns=['user','q','s_est','s_unc','s_rand'])
    i = 0
    for user in users:
        if missing[user] <= threshold:
            entropy_q.loc[i] = [user,missing[user],estimated_entropy_lz_suffix_array(sequences[user]),uncorrelated_entropy(sequences[user]),random_entropy(sequences[user])]
            i = i + 1
    return entropy_q

from scipy import optimize
import numpy as np
from functools import partial

def calc_p_max(sequences,entropy_q_threshold):
    pmax_q_threshold = pd.DataFrame(columns = ['user','PMax_est','PMax_unc','PMax_rand','PMax_exp'])
    for i in range(len(entropy_q_threshold)):
        row = entropy_q_threshold.iloc[i]
        #print(row)
        N = len(unique(sequences[row['user']]))
        #print(N)
        def f(x,s = 0.0 ):
            return ((-1 * (x * np.log2(x) + (1-x) * np.log2(1-x)) + (1-x) * np.log2(N-1)) - s)
        f_s_unc = partial(f, s = row['s_unc'])
        f_s_est = partial(f, s = row['s_est'])
        f_s_rand = partial(f, s = row['s_rand'])
        #f_s_exp = partial(f, s = row['s_exp'])
        xs = arange(0.001,1.0,0.05)
        f_x = [0] * len(xs)
        for j in arange(len(xs)):
            f_x[j] = f(xs[j],s = row['s_est'])
        #print f_x
        min_unc = f(0.001,s = row['s_unc'])
        max_unc = f(1.0,s = row['s_unc'])
        
        min_rand = f(0.001,s = row['s_rand'])
        max_rand = f(1.0,s = row['s_rand'])
        
        #print f_x[0],f_x[len(xs)-1]
        try:
            pmax_est = optimize.newton(f_s_est,0.5,maxiter=50)
        except:    
            if abs(f_x[0]) < abs(f_x[len(xs)-1]):
                pmax_est = 0.0
            else:
                pmax_est = 1.0
              
        #print min_unc,max_unc
        try:
            pmax_unc = optimize.newton(f_s_unc,0.5,maxiter=50)
        except:    
            if abs(min_unc) < abs(max_unc):
                pmax_unc = 0.0
            else:
                pmax_unc = 1.0
                
        #print min_rand,max_rand
        try:
            pmax_rand = optimize.newton(f_s_rand,0.5,maxiter=50)
        except:    
            if abs(min_rand) < abs(max_rand):
                pmax_rand = 0.0
            else:
                pmax_rand= 1.0

    pylab.xlabel("PMax_exp")
    pylab.ylabel("Equation")
    pylab.title("Evaluating the equation for possible values of predictability (PMax_est) for different users")
    return pmax_q_threshold
